\begin{frame}{V-B.1 Texture-aware Branch}
  % \begin{columns}
    % \column{0.55\textwidth}
      Từ đặc trưng không gian $F_{sp}$:
      \begin{align*}
        T &= \phi_\text{tex}(F_{sp}) \\
        F_{sp}^\text{tex} &= F_{sp} + \psi([F_{sp},\,T])
      \end{align*}
      \begin{itemize}
        \item Trích xuất: cạnh nhỏ, tương phản cục bộ, cấu trúc bề mặt
        \item $F_{sp}^\text{tex}$ đưa vào FFM cùng Context Path
      \end{itemize}
    % \column{0.42\textwidth}
      \begin{block}{Lỗi nhắm đến}
        Nhầm lẫn các lớp có bề mặt gần giống: thịt, bánh, sốt, rau chế biến
      \end{block}
  % \end{columns}
\end{frame}

\begin{frame}{V-B.2 Relation GNN Module}
  \textbf{Class prototype} (soft-mask pooling) + \textbf{Ma trận cạnh} từ thống kê tập train:
  \[
    f_i = \frac{\sum_u q_i(u)F(u)}{\sum_u q_i(u)+\varepsilon}, \quad
    A = \text{row\_norm}(\alpha A_\text{cooc} + \beta A_\text{touch} + \gamma A_\text{overlap})
  \]
  GNN cập nhật và sinh relational bias hiệu chỉnh logits:
  \[
    h_i^{(l+1)} = \sigma\!\Bigl(\sum_{j\in\mathcal{N}(i)} A_{ij}W^{(l)}h_j^{(l)}\Bigr), \quad
    Y'_i(u) = Y_i(u) + \eta b_i
  \]
  \begin{block}{Lỗi nhắm đến}
    Sink class, nhầm lẫn có cấu trúc theo đồng xuất hiện
  \end{block}
\end{frame}

\begin{frame}{V-B.3 Boundary-aware Supervision}
  % \begin{columns}
  %   \column{0.56\textwidth}
      Sinh bản đồ biên: $B = \text{Dilate}(S) - \text{Erode}(S)$

      Loss biên:
      \[
        L_\text{edge} = \text{WBCE}(\hat{B},B) + \lambda_d L_\text{Dice}(\hat{B},B)
      \]
      \begin{itemize}
        \item Tập trung sửa lỗi \textbf{mask tràn} và \textbf{biên dính lớp}.
        \item Boundary Head chỉ dùng khi train nên không làm chậm inference.
      \end{itemize}
    % \column{0.4\textwidth}
      \begin{block}{Lợi ích kỳ vọng}
        \begin{itemize}
          \item tăng IoU cho lớp mỏng/nhỏ
          \item giảm false positive quanh rìa món ăn
        \end{itemize}
      \end{block}
  % \end{columns}
\end{frame}
