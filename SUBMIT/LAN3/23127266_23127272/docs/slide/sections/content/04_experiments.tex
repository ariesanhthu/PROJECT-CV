\section{Thực nghiệm}

\begin{frame}{IV-A. Thiết lập huấn luyện baseline}
  \begin{block}{Dữ liệu huấn luyện: \texttt{FoodSeg103} nguyên thủy (103 lớp)}
    Baseline giữ nguyên không gian nhãn 103 lớp.
  \end{block}
  \vspace{2em}
  \begin{itemize}
    \item Tập test: 2.135 ảnh --- giữ nguyên để đánh giá cuối cùng
    \item Từ tập train gốc: chia train/val theo tỉ lệ 90/10 (fixed \texttt{random\_state})
    \item Ba vai trò tách biệt: \texttt{train} (tối ưu) $|$ \texttt{val} (theo dõi hội tụ) $|$ \texttt{test} (độc lập)
  \end{itemize}
  \vspace{0.4em}
  % \begin{alertblock}{Lưu ý về \texttt{FoodSeg103\_rebalanced}}
  %   Phiên bản 75 lớp (rebalanced) \textbf{không} được dùng để huấn luyện baseline. Đây là nhánh thực nghiệm bổ sung độc lập, được thực hiện \emph{sau} khi baseline đã được train và đánh giá xong trên dữ liệu gốc, nhằm phân tích thêm ảnh hưởng của phân mảnh nhãn.
  % \end{alertblock}
\end{frame}

\begin{frame}{IV-C. Nhận xét lỗi có cấu trúc}
  \begin{alertblock}{Hiện tượng Sink Class}
    \textbf{chicken duck} chiếm $\approx$458M pixel False Positive. Các lớp tương đồng về vân như (58.5\%), pork (57.9\%), bread (56.2\%), steak (55.6\%) đều bị dự đoán sai thành lớp trên.
  \end{alertblock}
  \vspace{0.3em}
  \textbf{Nhầm lẫn dựa trên đặc điểm thị giác}
  \begin{itemize}
    \item rice $\leftrightarrow$ potato (21.9\% / 20.1\%)
    \item steak $\to$ sausage (27.0\%), broccoli $\to$ sausage (27.6\%)
  \end{itemize}
  \vspace{0.3em}
  \begin{block}{Kết luận}
    Lỗi \textbf{không ngẫu nhiên}: do tương đồng về vân và quan hệ tương đồng xuất hiện. Spatial Path chưa đủ để phân biệt fine-grained texture.
  \end{block}
\end{frame}
