\section{Kết luận}

\begin{frame}{Kết luận}
    \begin{block}{Ba thông điệp cần chốt khi thuyết trình}
        \begin{enumerate}
            \item \textbf{Dimension reduction} giúp nhìn đúng cấu trúc dữ liệu: PCA mạnh về diễn giải, còn MDS/Isomap/t-SNE mạnh về manifold và trực quan hóa.
            \item \textbf{Linear classification} là baseline rất mạnh: Fisher/LDA, logistic regression và linear SVM đều dựa trên ý tưởng chiếu dữ liệu lên một trục phân biệt.
            \item \textbf{Non-linear learning} mở rộng sức biểu diễn của mô hình: kernel SVM, Random Forest, Gradient Boosting phù hợp khi biên quyết định không còn tuyến tính.
        \end{enumerate}
    \end{block}

    \begin{alertblock}{Nguyên tắc chọn mô hình}
        Bắt đầu từ cấu trúc hình học của dữ liệu, sau đó kiểm chứng bằng \textbf{cross-validation}, \textbf{balanced accuracy}, \textbf{ROC-AUC} và khả năng tổng quát hóa trên tập ngoài mẫu.
    \end{alertblock}

    \hfill{\scriptsize Tài liệu nền tảng: \cite{duchesnay2021statsml}}
\end{frame}
